// Fireball
actor Fireball
{
	Radius 6
	Height 8
	Speed 10
	FastSpeed 20
	
	Damage (ACS_NamedExecuteWithResult("SetProjectileDamage", 0)) // Energy * 0.5
	
	Projectile
	+RANDOMIZE
	
	RenderStyle Add
	Alpha 1
	
	SeeSound "imp/attack"
	DeathSound "imp/shotx"
	
	States
	{
	Spawn:
		BAL1 AB 4 Bright
		Loop
	Death:
		BAL1 CDE 6 Bright
		Stop
	}
}
