// Generic Item Dropper
actor DRPGItemDropper
{
    States
    {
    Spawn:
        TNT1 A 0 NoDelay A_Jump(32, "DropItem")
        TNT1 A -1
        Stop
    DropItem:
        TNT1 A 1 A_SpawnItemEx("DRPGGenericDropper", 0, 0, 0, Random(-4, 4), Random(-4, 4), Random(1, 8))
        TNT1 A 0 A_Jump(32, "DropItem")
        TNT1 A -1
        Stop
    }
}
